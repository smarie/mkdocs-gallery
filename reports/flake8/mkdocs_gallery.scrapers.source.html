76 def _matplotlib_fig_titles(fig):
77 titles = []
78 # get supertitle if exists
79 suptitle = getattr(fig, "_suptitle", None)
80 if suptitle is not None:
81 titles.append(suptitle.get_text())
82 # get titles from all axes, for all locs
83 title_locs = ["left", "center", "right"]
84 for ax in fig.axes:
85 for loc in title_locs:
86 text = ax.get_title(loc=loc)
87 if text:
88 titles.append(text)
89 fig_titles = ", ".join(titles)
90 return fig_titles

100 def matplotlib_scraper(block, script: GalleryScript, **kwargs):
101 """Scrape Matplotlib images.
102  
103 Parameters
104 ----------
105 block : tuple
106 A tuple containing the (label, content, line_number) of the block.
107  
108 script : GalleryScript
109 Dict of block variables.
110  
111 **kwargs : dict
112 Additional keyword arguments to pass to
113 :meth:`~matplotlib.figure.Figure.savefig`, e.g. ``format='svg'``.
114 The ``format`` kwarg in particular is used to set the file extension
115 of the output file (currently only 'png', 'jpg', and 'svg' are
116 supported).
117  
118 Returns
119 -------
120 md : str
121 The Markdown that will be rendered to HTML containing
122 the images. This is often produced by :func:`figure_md_or_html`.
123 """
124 try:
125 matplotlib, plt = _import_matplotlib()
126 except ImportError:
127 # Matplotlib is not installed. Ignore
128 # Note: we should better remove this (and the same in _reset_matplotlib)
129 # and auto-adjust the corresponding config option defaults (image_scrapers, reset_modules) when
130 # matplotlib is not present
131 return ""
132  
133 gallery_conf = script.gallery_conf
134 from matplotlib.animation import Animation
135  
136 image_mds = []
137  
138 # Check for srcset hidpi images
139 srcset = gallery_conf.get("image_srcset", [])
140 srcset_mult_facs = [1] # one is always supplied...
141 for st in srcset:
142 if (len(st) > 0) and (st[-1] == "x"):
143 # "2x" = "2.0"
144 srcset_mult_facs += [float(st[:-1])]
145 elif st == "":
146 pass
147 else:
148 raise ExtensionError(
149 f'Invalid value for image_srcset parameter: "{st}". '
150 "Must be a list of strings with the multiplicative "
151 'factor followed by an "x". e.g. ["2.0x", "1.5x"]'
152 )
153  
154 # Check for animations
155 anims = list()
156 if gallery_conf.get("matplotlib_animations", False):
157 for ani in script.run_vars.example_globals.values():
158 if isinstance(ani, Animation):
159 anims.append(ani)
160  
161 # Then standard images
162 for fig_num, image_path in zip(plt.get_fignums(), script.run_vars.image_path_iterator):
163 image_path = PurePosixPath(image_path)
164 if "format" in kwargs:
165 image_path = image_path.with_suffix("." + kwargs["format"])
166  
167 # Set the fig_num figure as the current figure as we can't save a figure that's not the current figure.
168 fig = plt.figure(fig_num)
169  
170 # Deal with animations
171 cont = False
172 for anim in anims:
173 if anim._fig is fig:
174 image_mds.append(_anim_md(anim, str(image_path), gallery_conf))
175 cont = True
176 break
177 if cont:
178 continue
179  
180 # get fig titles
181 fig_titles = _matplotlib_fig_titles(fig)
182 to_rgba = matplotlib.colors.colorConverter.to_rgba
183  
184 # shallow copy should be fine here, just want to avoid changing
185 # "kwargs" for subsequent figures processed by the loop
186 these_kwargs = kwargs.copy()
187 for attr in ["facecolor", "edgecolor"]:
188 fig_attr = getattr(fig, "get_" + attr)()
189 default_attr = matplotlib.rcParams["figure." + attr]
190 if to_rgba(fig_attr) != to_rgba(default_attr) and attr not in kwargs:
191 these_kwargs[attr] = fig_attr
192  
193 # save the figures, and populate the srcsetpaths
194 try:
195 fig.savefig(image_path, **these_kwargs)
196 dpi0 = matplotlib.rcParams["savefig.dpi"]
197 if dpi0 == "figure":
198 dpi0 = fig.dpi
199 dpi0 = these_kwargs.get("dpi", dpi0)
200 srcsetpaths = {0: image_path}
201  
202 # save other srcset paths, keyed by multiplication factor:
203 for mult in srcset_mult_facs:
204 if not (mult == 1):
205 multst = f"{mult}".replace(".", "_")
206 name = f"{image_path.stem}_{multst}x{image_path.suffix}"
207 hipath = image_path.parent / PurePosixPath(name)
208 hikwargs = these_kwargs.copy()
209 hikwargs["dpi"] = mult * dpi0
210 fig.savefig(hipath, **hikwargs)
211 srcsetpaths[mult] = hipath
212 srcsetpaths = [srcsetpaths]
213 except Exception:
214 plt.close("all")
215 raise
216  
217 if "images" in gallery_conf["compress_images"]:
218 optipng(str(image_path), gallery_conf["compress_images_args"])
219 for hipath in srcsetpaths[0].items():
220 optipng(str(hipath), gallery_conf["compress_images_args"])
221  
222 image_mds.append((image_path, fig_titles, srcsetpaths))
223  
224 plt.close("all")
225  
226 # Create the markdown or html output
227 # <li>
228 # <img src="../_images/mkd_glr_plot_1_exp_001.png"
229 # srcset="../_images/mkd_glr_plot_1_exp_001.png, ../_images/mkd_glr_plot_1_exp_001_2_0x.png 2.0x"
230 # alt="Exponential function" class="sphx-glr-multi-img">
231 # </li>
232 # <li>
233 # <img src="../_images/mkd_glr_plot_1_exp_002.png"
234 # srcset="../_images/mkd_glr_plot_1_exp_002.png, ../_images/mkd_glr_plot_1_exp_002_2_0x.png 2.0x"
235 # alt="Negative exponential function" class="sphx-glr-multi-img">
236 # </li>
237  
238 md = ""
239 if len(image_mds) == 1:
240 if isinstance(image_mds[0], str):
241 # an animation, see _anim_md
242 md = image_mds[0]
243 else:
244 # an image
245 image_path, fig_titles, srcsetpaths = image_mds[0]
246 md = figure_md_or_html([image_path], script, fig_titles, srcsetpaths=srcsetpaths)
247 elif len(image_mds) > 1:
248 # Old
249 # Replace the 'single' CSS class by the 'multi' one
250 # image_mds = [re.sub(r"mkd-glr-single-img", "mkd-glr-multi-img", image) for image in image_mds]
251 # image_mds = [HLIST_IMAGE_MATPLOTLIB % image for image in image_mds]
252 # md = HLIST_HEADER % (''.join(image_mds))
253  
254 # New: directly use the html
255 image_htmls = []
256 for image_path, fig_titles, srcsetpaths in image_mds:
257 img_html = figure_md_or_html([image_path], script, fig_titles, srcsetpaths=srcsetpaths, raw_html=True)
258 image_htmls.append(img_html)
259 md = HLIST_HEADER % ("".join(image_htmls))
260 return md

263 def _anim_md(anim, image_path, gallery_conf):
264 import matplotlib
265 from matplotlib.animation import FFMpegWriter, ImageMagickWriter
266  
267 # output the thumbnail as the image, as it will just be copied
268 # if it's the file thumbnail
269 fig = anim._fig
270 image_path = image_path.replace(".png", ".gif")
271 fig_size = fig.get_size_inches()
272 thumb_size = gallery_conf["thumbnail_size"]
273 use_dpi = round(min(t_s / f_s for t_s, f_s in zip(thumb_size, fig_size)))
274 # FFmpeg is buggy for GIFs before Matplotlib 3.3.1
275 if LooseVersion(matplotlib.__version__) >= LooseVersion("3.3.1") and FFMpegWriter.isAvailable():
276 writer = "ffmpeg"
277 elif ImageMagickWriter.isAvailable():
278 writer = "imagemagick"
279 else:
280 writer = None
281 anim.save(image_path, writer=writer, dpi=use_dpi)
282 html = anim._repr_html_()
283 if html is None: # plt.rcParams['animation.html'] == 'none'
284 html = anim.to_jshtml()
285 html = indent(html, " ")
286 return _ANIMATION_RST.format(html=html)

289 def mayavi_scraper(block, script: GalleryScript):
290 """Scrape Mayavi images.
291  
292 Parameters
293 ----------
294 block : tuple
295 A tuple containing the (label, content, line_number) of the block.
296  
297 script : GalleryScript
298 Script being run
299  
300 Returns
301 -------
302 md : str
303 The ReSTructuredText that will be rendered to HTML containing
304 the images. This is often produced by :func:`figure_md_or_html`.
305 """
306 from mayavi import mlab
307  
308 image_path_iterator = script.run_vars.image_path_iterator
309 image_paths = list()
310 e = mlab.get_engine()
311 for scene, image_path in zip(e.scenes, image_path_iterator):
312 try:
313 mlab.savefig(image_path, figure=scene)
314 except Exception:
315 mlab.close(all=True)
316 raise
317 # make sure the image is not too large
318 rescale_image(image_path, image_path, 850, 999)
319 if "images" in script.gallery_conf["compress_images"]:
320 optipng(image_path, script.gallery_conf["compress_images_args"])
321 image_paths.append(image_path)
322 mlab.close(all=True)
323 return figure_md_or_html(image_paths, script)

344 def _find_image_ext(path: Path, raise_if_not_found: bool = True) -> Path:
345 """Find an image, tolerant of different file extensions."""
346  
347 for ext in _KNOWN_IMG_EXTS:
348 this_path = path.with_suffix(ext)
349 if this_path.exists():
350 break
351 else:
352 if raise_if_not_found:
353 raise ImageNotFoundError(path)
354  
355 # None exists. Default to png.
356 ext = ".png"
357 this_path = path.with_suffix(ext)
358  
359 return this_path, ext

362 def save_figures(block, script: GalleryScript):
363 """Save all open figures of the example code-block.
364  
365 Parameters
366 ----------
367 block : tuple
368 A tuple containing the (label, content, line_number) of the block.
369  
370 script : GalleryScript
371 Script run.
372  
373 Returns
374 -------
375 images_md : str
376 md code to embed the images in the document.
377 """
378 image_path_iterator = script.run_vars.image_path_iterator
379 all_md = ""
380 prev_count = len(image_path_iterator)
381 for scraper in script.gallery_conf["image_scrapers"]:
382 # Use the scraper to generate the md containing image(s) (may be several)
383 md = scraper(block, script)
384 if not isinstance(md, str):
385 raise ExtensionError(f"md from scraper {scraper!r} was not a string, got type {type(md)}:\n{md!r}")
386  
387 # Make sure that all images generated by the scraper exist.
388 n_new = len(image_path_iterator) - prev_count
389 for ii in range(n_new):
390 current_path, ext = _find_image_ext(image_path_iterator.paths[prev_count + ii])
391 if not current_path.exists():
392 raise ExtensionError(f"Scraper {scraper!r} did not produce expected image:\n{current_path}")
393  
394 all_md += md
395  
396 return all_md

402 def figure_md_or_html(
403 figure_paths: List[Path],
404 script: GalleryScript,
405 fig_titles: str = "",
406 srcsetpaths: List[Dict[float, Path]] = None,
407 raw_html=False,
408 ):
409 """Generate md or raw html for a list of image filenames.
410  
411 Depending on whether we have one or more figures, we use a
412 single md call to 'image' or a horizontal list.
413  
414 Parameters
415 ----------
416 figure_paths : List[Path]
417 List of strings of the figures' absolute paths.
418 sources_dir : Path
419 absolute path of Sphinx documentation sources
420 fig_titles : str
421 Titles of figures, empty string if no titles found. Currently
422 only supported for matplotlib figures, default = ''.
423 srcsetpaths : list or None
424 List of dictionaries containing absolute paths. If
425 empty, then srcset field is populated with the figure path.
426 (see ``image_srcset`` configuration option). Otherwise,
427 each dict is of the form
428 {0: /images/image.png, 2.0: /images/image_2_0x.png}
429 where the key is the multiplication factor and the contents
430 the path to the image created above.
431  
432 Returns
433 -------
434 images_md : str
435 md code to embed the images in the document
436  
437 The md code will have a custom ``image-sg`` directive that allows
438 multiple resolution images to be served e.g.:
439 ``:srcset: /plot_types/imgs/img_001.png,
440 /plot_types/imgs/img_2_0x.png 2.0x``
441  
442 """
443  
444 if srcsetpaths is None:
445 # this should never happen, but figure_md_or_html is public, so
446 # this has to be a kwarg...
447 srcsetpaths = [{0: fl} for fl in figure_paths]
448  
449 # Get all images relative to the website sources root
  • F841 Local variable 'sources_dir' is assigned to but never used
450 sources_dir = script.gallery.all_info.mkdocs_docs_dir
451 script_md_dir = script.gallery.generated_dir
452  
453 # Get alt text
454 alt = ""
455 if fig_titles:
456 alt = fig_titles
457 elif figure_paths:
458 file_name = os.path.split(str(figure_paths[0]))[1]
459 # remove ext & 'mkd_glr_' from start & n#'s from end
460 file_name_noext = os.path.splitext(file_name)[0][PREFIX_LEN:-4]
461 # replace - & _ with \s
462 file_name_final = re.sub(r"[-,_]", " ", file_name_noext)
463 alt = file_name_final
464  
465 alt = _single_line_sanitize(alt)
466  
467 images_md = ""
468 if len(figure_paths) == 1:
469 figure_path = figure_paths[0]
470 hinames = srcsetpaths[0]
471 srcset = _get_srcset_st(script_md_dir, hinames)
472 figure_path_rel_to_script_md_dir = figure_path.relative_to(script_md_dir).as_posix().lstrip("/")
473 if raw_html:
474 # html version
475 images_md = (
476 f'<img alt="{alt}" src="../{figure_path_rel_to_script_md_dir}" srcset="{srcset}", '
477 f'class="sphx-glr-single-img" />'
478 )
479 else:
480 # markdown version
481 images_md = f'![{alt}](./{figure_path_rel_to_script_md_dir}){{: .mkd-glr-single-img srcset="{srcset}"}}'
482  
483 elif len(figure_paths) > 1:
484 images_md = HLIST_HEADER
485 for nn, figure_path in enumerate(figure_paths):
486 hinames = srcsetpaths[nn]
487 srcset = _get_srcset_st(script_md_dir, hinames)
488 figure_path_rel_to_script_md_dir = figure_path.relative_to(script_md_dir).as_posix().lstrip("/")
489 images_md += HLIST_SG_TEMPLATE % (
490 alt,
491 figure_path_rel_to_script_md_dir,
492 srcset,
493 )
494  
495 return images_md

498 def _get_srcset_st(sources_dir: Path, hinames: Dict[float, Path]):
499 """
500 Create the srcset string for including on the md line.
501 ie. sources_dir might be /home/sample-proj/source,
502 hinames posix paths to
503 0: /home/sample-proj/source/plot_types/images/img1.png,
504 2.0: /home/sample-proj/source/plot_types/images/img1_2_0x.png,
505 The result will be:
506 '/plot_types/basic/images/mkd_glr_pie_001.png,
507 /plot_types/basic/images/mkd_glr_pie_001_2_0x.png 2.0x'
508 """
509 srcst = ""
510 for k in hinames.keys():
511 path = hinames[k].relative_to(sources_dir).as_posix().lstrip("/")
512 srcst += "../" + path
513 if k == 0:
514 srcst += ", "
515 else:
516 srcst += f" {k:1.1f}x, "
517 if srcst[-2:] == ", ":
518 srcst = srcst[:-2]
519 srcst += ""
520  
521 return srcst
