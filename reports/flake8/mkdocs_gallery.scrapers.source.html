67 def _matplotlib_fig_titles(fig):
68 titles = []
69 # get supertitle if exists
70 suptitle = getattr(fig, "_suptitle", None)
71 if suptitle is not None:
72 titles.append(suptitle.get_text())
73 # get titles from all axes, for all locs
74 title_locs = ['left', 'center', 'right']
75 for ax in fig.axes:
76 for loc in title_locs:
77 text = ax.get_title(loc=loc)
78 if text:
79 titles.append(text)
80 fig_titles = ', '.join(titles)
81 return fig_titles

91 def matplotlib_scraper(block, script: GalleryScript, **kwargs):
92 """Scrape Matplotlib images.
93  
94 Parameters
95 ----------
96 block : tuple
97 A tuple containing the (label, content, line_number) of the block.
98  
99 script : GalleryScript
100 Dict of block variables.
101  
102 **kwargs : dict
103 Additional keyword arguments to pass to
104 :meth:`~matplotlib.figure.Figure.savefig`, e.g. ``format='svg'``.
105 The ``format`` kwarg in particular is used to set the file extension
106 of the output file (currently only 'png', 'jpg', and 'svg' are
107 supported).
108  
109 Returns
110 -------
111 md : str
112 The Markdown that will be rendered to HTML containing
113 the images. This is often produced by :func:`figure_md_or_html`.
114 """
115 try:
116 matplotlib, plt = _import_matplotlib()
117 except ImportError:
118 # Matplotlib is not installed. Ignore
119 # Note: we should better remove this (and the same in _reset_matplotlib)
120 # and auto-adjust the corresponding config option defaults (image_scrapers, reset_modules) when
121 # matplotlib is not present
122 return ""
123  
124 gallery_conf = script.gallery_conf
125 from matplotlib.animation import Animation
126  
127 image_mds = []
128  
129 # Check for srcset hidpi images
130 srcset = gallery_conf.get('image_srcset', [])
131 srcset_mult_facs = [1] # one is always supplied...
132 for st in srcset:
133 if (len(st) > 0) and (st[-1] == 'x'):
134 # "2x" = "2.0"
135 srcset_mult_facs += [float(st[:-1])]
136 elif st == "":
137 pass
138 else:
139 raise ExtensionError(
140 f'Invalid value for image_srcset parameter: "{st}". '
141 'Must be a list of strings with the multiplicative '
142 'factor followed by an "x". e.g. ["2.0x", "1.5x"]')
143  
144 # Check for animations
145 anims = list()
146 if gallery_conf.get('matplotlib_animations', False):
147 for ani in script.run_vars.example_globals.values():
148 if isinstance(ani, Animation):
149 anims.append(ani)
150  
151 # Then standard images
152 for fig_num, image_path in zip(plt.get_fignums(), script.run_vars.image_path_iterator):
153 image_path = PurePosixPath(image_path)
154 if 'format' in kwargs:
155 image_path = image_path.with_suffix('.' + kwargs['format'])
156  
157 # Set the fig_num figure as the current figure as we can't save a figure that's not the current figure.
158 fig = plt.figure(fig_num)
159  
160 # Deal with animations
161 cont = False
162 for anim in anims:
163 if anim._fig is fig:
164 image_mds.append(_anim_md(anim, str(image_path), gallery_conf))
165 cont = True
166 break
167 if cont:
168 continue
169  
170 # get fig titles
171 fig_titles = _matplotlib_fig_titles(fig)
172 to_rgba = matplotlib.colors.colorConverter.to_rgba
173  
174 # shallow copy should be fine here, just want to avoid changing
175 # "kwargs" for subsequent figures processed by the loop
176 these_kwargs = kwargs.copy()
177 for attr in ['facecolor', 'edgecolor']:
178 fig_attr = getattr(fig, 'get_' + attr)()
179 default_attr = matplotlib.rcParams['figure.' + attr]
180 if to_rgba(fig_attr) != to_rgba(default_attr) and \
181 attr not in kwargs:
182 these_kwargs[attr] = fig_attr
183  
184 # save the figures, and populate the srcsetpaths
185 try:
186 fig.savefig(image_path, **these_kwargs)
187 dpi0 = matplotlib.rcParams['savefig.dpi']
188 if dpi0 == 'figure':
189 dpi0 = fig.dpi
190 dpi0 = these_kwargs.get('dpi', dpi0)
191 srcsetpaths = {0: image_path}
192  
193 # save other srcset paths, keyed by multiplication factor:
194 for mult in srcset_mult_facs:
195 if not (mult == 1):
196 multst = f'{mult}'.replace('.', '_')
197 name = f"{image_path.stem}_{multst}x{image_path.suffix}"
198 hipath = image_path.parent / PurePosixPath(name)
199 hikwargs = these_kwargs.copy()
200 hikwargs['dpi'] = mult * dpi0
201 fig.savefig(hipath, **hikwargs)
202 srcsetpaths[mult] = hipath
203 srcsetpaths = [srcsetpaths]
204 except Exception:
205 plt.close('all')
206 raise
207  
208 if 'images' in gallery_conf['compress_images']:
209 optipng(str(image_path), gallery_conf['compress_images_args'])
210 for hipath in srcsetpaths[0].items():
211 optipng(str(hipath), gallery_conf['compress_images_args'])
212  
213 image_mds.append((image_path, fig_titles, srcsetpaths))
214  
215 plt.close('all')
216  
217 # Create the markdown or html output
218 # <li>
219 # <img src="../_images/mkd_glr_plot_1_exp_001.png"
220 # srcset="../_images/mkd_glr_plot_1_exp_001.png, ../_images/mkd_glr_plot_1_exp_001_2_0x.png 2.0x"
221 # alt="Exponential function" class="sphx-glr-multi-img">
222 # </li>
223 # <li>
224 # <img src="../_images/mkd_glr_plot_1_exp_002.png"
225 # srcset="../_images/mkd_glr_plot_1_exp_002.png, ../_images/mkd_glr_plot_1_exp_002_2_0x.png 2.0x"
226 # alt="Negative exponential function" class="sphx-glr-multi-img">
227 # </li>
228  
229 md = ''
230 if len(image_mds) == 1:
231 if isinstance(image_mds[0], str):
232 # an animation, see _anim_md
233 md = image_mds[0]
234 else:
235 # an image
236 image_path, fig_titles, srcsetpaths = image_mds[0]
237 md = figure_md_or_html([image_path], script, fig_titles, srcsetpaths=srcsetpaths)
238 elif len(image_mds) > 1:
239 # Old
240 # Replace the 'single' CSS class by the 'multi' one
241 # image_mds = [re.sub(r"mkd-glr-single-img", "mkd-glr-multi-img", image) for image in image_mds]
242 # image_mds = [HLIST_IMAGE_MATPLOTLIB % image for image in image_mds]
243 # md = HLIST_HEADER % (''.join(image_mds))
244  
245 # New: directly use the html
246 image_htmls = []
247 for image_path, fig_titles, srcsetpaths in image_mds:
248 img_html = figure_md_or_html([image_path], script, fig_titles, srcsetpaths=srcsetpaths, raw_html=True)
249 image_htmls.append(img_html)
250 md = HLIST_HEADER % (''.join(image_htmls))
251 return md

254 def _anim_md(anim, image_path, gallery_conf):
255 import matplotlib
256 from matplotlib.animation import FFMpegWriter, ImageMagickWriter
257 # output the thumbnail as the image, as it will just be copied
258 # if it's the file thumbnail
259 fig = anim._fig
260 image_path = image_path.replace('.png', '.gif')
261 fig_size = fig.get_size_inches()
262 thumb_size = gallery_conf['thumbnail_size']
263 use_dpi = round(
264 min(t_s / f_s for t_s, f_s in zip(thumb_size, fig_size)))
265 # FFmpeg is buggy for GIFs before Matplotlib 3.3.1
266 if LooseVersion(matplotlib.__version__) >= LooseVersion('3.3.1') and \
267 FFMpegWriter.isAvailable():
268 writer = 'ffmpeg'
269 elif ImageMagickWriter.isAvailable():
270 writer = 'imagemagick'
271 else:
272 writer = None
273 anim.save(image_path, writer=writer, dpi=use_dpi)
274 html = anim._repr_html_()
275 if html is None: # plt.rcParams['animation.html'] == 'none'
276 html = anim.to_jshtml()
277 html = indent(html, ' ')
278 return _ANIMATION_RST.format(html=html)

281 def mayavi_scraper(block, script: GalleryScript):
282 """Scrape Mayavi images.
283  
284 Parameters
285 ----------
286 block : tuple
287 A tuple containing the (label, content, line_number) of the block.
288  
289 script : GalleryScript
290 Script being run
291  
292 Returns
293 -------
294 md : str
295 The ReSTructuredText that will be rendered to HTML containing
296 the images. This is often produced by :func:`figure_md_or_html`.
297 """
298 from mayavi import mlab
299 image_path_iterator = script.run_vars.image_path_iterator
300 image_paths = list()
301 e = mlab.get_engine()
302 for scene, image_path in zip(e.scenes, image_path_iterator):
303 try:
304 mlab.savefig(image_path, figure=scene)
305 except Exception:
306 mlab.close(all=True)
307 raise
308 # make sure the image is not too large
309 rescale_image(image_path, image_path, 850, 999)
310 if 'images' in script.gallery_conf['compress_images']:
311 optipng(image_path, script.gallery_conf['compress_images_args'])
312 image_paths.append(image_path)
313 mlab.close(all=True)
314 return figure_md_or_html(image_paths, script)

335 def _find_image_ext(path: Path, raise_if_not_found: bool = True) -> Path:
336 """Find an image, tolerant of different file extensions."""
337  
338 for ext in _KNOWN_IMG_EXTS:
339 this_path = path.with_suffix(ext)
340 if this_path.exists():
341 break
342 else:
343 if raise_if_not_found:
344 raise ImageNotFoundError(path)
345  
346 # None exists. Default to png.
347 ext = '.png'
348 this_path = path.with_suffix(ext)
349  
350 return this_path, ext

353 def save_figures(block, script: GalleryScript):
354 """Save all open figures of the example code-block.
355  
356 Parameters
357 ----------
358 block : tuple
359 A tuple containing the (label, content, line_number) of the block.
360  
361 script : GalleryScript
362 Script run.
363  
364 Returns
365 -------
366 images_md : str
367 md code to embed the images in the document.
368 """
369 image_path_iterator = script.run_vars.image_path_iterator
370 all_md = u''
371 prev_count = len(image_path_iterator)
372 for scraper in script.gallery_conf['image_scrapers']:
373 # Use the scraper to generate the md containing image(s) (may be several)
374 md = scraper(block, script)
375 if not isinstance(md, str):
376 raise ExtensionError(f"md from scraper {scraper!r} was not a string, got type {type(md)}:\n{md!r}")
377  
378 # Make sure that all images generated by the scraper exist.
379 n_new = len(image_path_iterator) - prev_count
380 for ii in range(n_new):
381 current_path, ext = _find_image_ext(image_path_iterator.paths[prev_count + ii])
382 if not current_path.exists():
383 raise ExtensionError(f"Scraper {scraper!r} did not produce expected image:\n{current_path}")
384  
385 all_md += md
386  
387 return all_md

393 def figure_md_or_html(
394 figure_paths: List[Path],
395 script: GalleryScript,
396 fig_titles: str = '',
397 srcsetpaths: List[Dict[float, Path]] = None,
398 raw_html=False
399 ):
400 """Generate md or raw html for a list of image filenames.
401  
402 Depending on whether we have one or more figures, we use a
403 single md call to 'image' or a horizontal list.
404  
405 Parameters
406 ----------
407 figure_paths : List[Path]
408 List of strings of the figures' absolute paths.
409 sources_dir : Path
410 absolute path of Sphinx documentation sources
411 fig_titles : str
412 Titles of figures, empty string if no titles found. Currently
413 only supported for matplotlib figures, default = ''.
414 srcsetpaths : list or None
415 List of dictionaries containing absolute paths. If
416 empty, then srcset field is populated with the figure path.
417 (see ``image_srcset`` configuration option). Otherwise,
418 each dict is of the form
419 {0: /images/image.png, 2.0: /images/image_2_0x.png}
420 where the key is the multiplication factor and the contents
421 the path to the image created above.
422  
423 Returns
424 -------
425 images_md : str
426 md code to embed the images in the document
427  
428 The md code will have a custom ``image-sg`` directive that allows
429 multiple resolution images to be served e.g.:
430 ``:srcset: /plot_types/imgs/img_001.png,
431 /plot_types/imgs/img_2_0x.png 2.0x``
432  
433 """
434  
435 if srcsetpaths is None:
436 # this should never happen, but figure_md_or_html is public, so
437 # this has to be a kwarg...
438 srcsetpaths = [{0: fl} for fl in figure_paths]
439  
440 # Get all images relative to the website sources root
  • F841 Local variable 'sources_dir' is assigned to but never used
441 sources_dir = script.gallery.all_info.mkdocs_docs_dir
442 script_md_dir = script.gallery.generated_dir
443  
444 # Get alt text
445 alt = ''
446 if fig_titles:
447 alt = fig_titles
448 elif figure_paths:
449 file_name = os.path.split(str(figure_paths[0]))[1]
450 # remove ext & 'mkd_glr_' from start & n#'s from end
451 file_name_noext = os.path.splitext(file_name)[0][PREFIX_LEN:-4]
452 # replace - & _ with \s
453 file_name_final = re.sub(r'[-,_]', ' ', file_name_noext)
454 alt = file_name_final
455  
456 alt = _single_line_sanitize(alt)
457  
458 images_md = ""
459 if len(figure_paths) == 1:
460 figure_path = figure_paths[0]
461 hinames = srcsetpaths[0]
462 srcset = _get_srcset_st(script_md_dir, hinames)
463 figure_path_rel_to_script_md_dir = figure_path.relative_to(script_md_dir).as_posix().lstrip('/')
464 if raw_html:
465 # html version
466 images_md = f'<img alt="{alt}" src="../{figure_path_rel_to_script_md_dir}" srcset="{srcset}", ' \
467 f'class="sphx-glr-single-img" />'
468 else:
469 # markdown version
470 images_md = f'![{alt}](./{figure_path_rel_to_script_md_dir}){{: .mkd-glr-single-img srcset="{srcset}"}}'
471  
472 elif len(figure_paths) > 1:
473 images_md = HLIST_HEADER
474 for nn, figure_path in enumerate(figure_paths):
475 hinames = srcsetpaths[nn]
476 srcset = _get_srcset_st(script_md_dir, hinames)
477 figure_path_rel_to_script_md_dir = figure_path.relative_to(script_md_dir).as_posix().lstrip('/')
478 images_md += (HLIST_SG_TEMPLATE % (alt, figure_path_rel_to_script_md_dir, srcset))
479  
480 return images_md

483 def _get_srcset_st(sources_dir: Path, hinames: Dict[float, Path]):
484 """
485 Create the srcset string for including on the md line.
486 ie. sources_dir might be /home/sample-proj/source,
487 hinames posix paths to
488 0: /home/sample-proj/source/plot_types/images/img1.png,
489 2.0: /home/sample-proj/source/plot_types/images/img1_2_0x.png,
490 The result will be:
491 '/plot_types/basic/images/mkd_glr_pie_001.png,
492 /plot_types/basic/images/mkd_glr_pie_001_2_0x.png 2.0x'
493 """
494 srcst = ''
495 for k in hinames.keys():
496 path = hinames[k].relative_to(sources_dir).as_posix().lstrip('/')
497 srcst += '../' + path
498 if k == 0:
499 srcst += ', '
500 else:
501 srcst += f' {k:1.1f}x, '
502 if srcst[-2:] == ', ':
503 srcst = srcst[:-2]
504 srcst += ''
505  
506 return srcst
