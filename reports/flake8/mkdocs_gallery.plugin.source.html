30 class ConfigList(co.OptionallyRequired):
31 """A list or single element of configuration matching a specific ConfigOption"""

33 def __init__(self, item_config: co.BaseConfigOption, single_elt_allowed: bool = True, **kwargs):
34 super().__init__(**kwargs)
35 self.single_elt_allowed = single_elt_allowed
36 self.item_config = item_config

38 def run_validation(self, value):
39 if not isinstance(value, (list, tuple)):
40 if self.single_elt_allowed:
41 value = (value, )
42 else:
43 msg = f"Expected a list but received a single element: {value}."
44 raise ValidationError(msg)
45  
46 # Validate all elements in the list
47 result = []
48 for i, v in enumerate(value):
49 try:
50 result.append(self.item_config.validate(v))
51 except ValidationError as e:
52 raise ValidationError(f"Error validating config item #{i+1}: {e}")
53 return result

56 class MySubConfig(co.SubConfig):
57 """Same as SubConfig except that it will be an empty dict when nothing is provided by user,
58 instead of a dict with all options containing their default values."""
59  
60 def validate(self, value):
61 if value is None or len(value) == 0:
62 return None
63 else:
64 return super(MySubConfig, self).validate(value)
65  
66 def run_validation(self, value):
67 """Fix SubConfig: errors and warnings were not caught
68  
69 See https://github.com/mkdocs/mkdocs/pull/2710
70 """
71 failed, self.warnings = Config.validate(self)
72 if len(failed) > 0:
73 # get the first failing one
74 key, err = failed[0]
75 raise ConfigurationError(f"Sub-option {key!r} configuration error: {err}")
76  
77 return self

258 # Add the gallery config script if needed
259 if self.conf_script:

262 if conf_script_parent == ".":

267 conf_script = conf_script.as_posix()

269 def exclude(i):
270 # Get a posix version of the relative path so as to be sure to match ok
271 posix_src_path = Path(i.src_path).as_posix()
272  
273 # Is it the conf script or a derived work of the conf script ?
274 if self.conf_script:
275 if posix_src_path == conf_script or conf_script_match.match(posix_src_path):
276 return True
277  
278 # Is it located in a gallery source directory ?
279 for d in examples_dirs:
280 if posix_src_path.startswith(d):
281 return True
282  
283 # Is it a binder dependency file ?
284 if posix_src_path in binder_files:
285 return True
286  
287 return False
288  
289 out = []
290 for i in files:
291 if not exclude(i):
292 out.append(i)
293  
294 return Files(out)

375 def merge_extra_config(extra_config: Dict[str, Any], config):
376 """Extend the configuration 'markdown_extensions' list with extension_name if needed."""
377  
378 for extension_cfg in extra_config["markdown_extensions"]:
379 if isinstance(extension_cfg, str):
380 extension_name = extension_cfg
381 if extension_name not in config['markdown_extensions']:
382 config['markdown_extensions'].append(extension_name)
383 elif isinstance(extension_cfg, dict):
384 assert len(extension_cfg) == 1 # noqa
385 extension_name, extension_options = extension_cfg.popitem()
386 if extension_name not in config['markdown_extensions']:
387 config['markdown_extensions'].append(extension_name)
388 if extension_name not in config['mdx_configs']:
389 config['mdx_configs'][extension_name] = extension_options
390 else:
391 # Only add options that are not already set
392 # TODO should we warn ?
393 for cfg_key, cfg_val in extension_options.items():
394 if cfg_key not in config['mdx_configs'][extension_name]:
395 config['mdx_configs'][extension_name][cfg_key] = cfg_val
396 else:
397 raise TypeError(extension_cfg)
