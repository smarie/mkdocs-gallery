59 class _LoggingTee(object):
60 """A tee object to redirect streams to the logger."""
61  
62 def __init__(self, src_filename: Path):
63 self.logger = logger
64 self.src_filename = src_filename
65 self.logger_buffer = ''
66 self.set_std_and_reset_position()
67  
68 def set_std_and_reset_position(self):
69 if not isinstance(sys.stdout, _LoggingTee):
70 self.origs = (sys.stdout, sys.stderr)
71 sys.stdout = sys.stderr = self
72 self.first_write = True
73 self.output = StringIO()
74 return self
75  
76 def restore_std(self):
77 sys.stdout.flush()
78 sys.stderr.flush()
79 sys.stdout, sys.stderr = self.origs
80  
81 def write(self, data):
82 self.output.write(data)
83  
84 if self.first_write:
85 self.logger.verbose('Output from %s', self.src_filename) # color='brown')
86 self.first_write = False
87  
88 data = self.logger_buffer + data
89 lines = data.splitlines()
90 if data and data[-1] not in '\r\n':
91 # Wait to write last line if it's incomplete. It will write next
92 # time or when the LoggingTee is flushed.
93 self.logger_buffer = lines[-1]
94 lines = lines[:-1]
95 else:
96 self.logger_buffer = ''
97  
98 for line in lines:
99 self.logger.verbose('%s', line)
100  
101 def flush(self):
102 self.output.flush()
103 if self.logger_buffer:
104 self.logger.verbose('%s', self.logger_buffer)
105 self.logger_buffer = ''

107 # When called from a local terminal seaborn needs it in Python3
108 def isatty(self):
109 return self.output.isatty()
110  
111 # When called in gen_single, conveniently use context managing
112 def __enter__(self):
113 return self
114  
115 def __exit__(self, type_, value, tb):
116 self.restore_std()

191 def _sanitize_md(string):
192 """Use regex to remove at least some sphinx directives.
193  
194 TODO is this still needed ?
195 """
196 # :class:`a.b.c <thing here>`, :ref:`abc <thing here>` --> thing here
197 p, e = r'(\s|^):[^:\s]+:`', r'`(\W|$)'
198 string = re.sub(p + r'\S+\s*<([^>`]+)>' + e, r'\1\2\3', string)
199 # :class:`~a.b.c` --> c
200 string = re.sub(p + r'~([^`]+)' + e, _regroup, string)
201 # :class:`a.b.c` --> a.b.c
202 string = re.sub(p + r'([^`]+)' + e, r'\1\2\3', string)
203  
204 # ``whatever thing`` --> whatever thing
205 p = r'(\s|^)`'
206 string = re.sub(p + r'`([^`]+)`' + e, r'\1\2\3', string)
207 # `whatever thing` --> whatever thing
208 string = re.sub(p + r'([^`]+)' + e, r'\1\2\3', string)
209 return string

212 # Find RST/Markdown title chars,
213 # i.e. lines that consist of (3 or more of the same) 7-bit non-ASCII chars.
214 # This conditional is not perfect but should hopefully be good enough.
215 RE_3_OR_MORE_NON_ASCII = r"([\W _])\1{3,}" # 3 or more identical chars
216  
217 RST_TITLE_MARKER = re.compile(rf'^[ ]*{RE_3_OR_MORE_NON_ASCII}[ ]*$')
218 MD_TITLE_MARKER = re.compile(r'^[ ]*[#]+[ ]*(.*)[ ]*$') # One or more starting hash with optional whitespaces before.

223 """Same as `extract_intro_and_title` for the readme files in galleries, but does not return the introduction.

246 def extract_readme_last_subtitle(file: Path, contents: str) -> str:
247 """Same as `extract_intro_and_title` for the readme files in galleries, but does not return the introduction.
248  
249 Parameters
250 ----------
251 file : Path
252 The readme file path (used for error messages only).
253  
254 contents : str
255 The already parsed readme contents
256  
257 Returns
258 -------
259 last_subtitle : str
260 The readme last title, or None.
261 """
262 paragraphs = extract_paragraphs(contents)
263  
264 # iterate from last paragraph
265 last_subtitle = None
266 for p in reversed(paragraphs):
267 current_is_good = False
268 for line in reversed(p.splitlines()):
269 if current_is_good:
270 last_subtitle = line
271 break
272 # Does this line contain a title ?
273 # - md style
274 md_match = MD_TITLE_MARKER.search(line)
275 if md_match:
276 last_subtitle = md_match.group(1)
277 break
278  
279 # - rst style
280 rst_match = RST_TITLE_MARKER.search(line)
281 if rst_match:
282 current_is_good = True
283  
284 if last_subtitle:
285 break
286  
287 return last_subtitle

300 def extract_intro_and_title(docstring: str, script: GalleryScript) -> Tuple[str, str]:
301 """Extract and clean the first paragraph of module-level docstring.
302  
303 The title is not saved in the `script` object in this process, users have to do it explicitly.
304  
305 Parameters
306 ----------
307 docstring : str
308 The docstring extracted from the top of the script.
309  
310 script : GalleryScript
311 The script where the docstring was extracted from (used for error messages only).
312  
313 Returns
314 -------
315 title : str
316 The title
317  
318 introduction : str
319 The introduction
320 """
321 # Extract paragraphs from the text
322 paragraphs = extract_paragraphs(docstring)
323 if len(paragraphs) == 0:
324 raise ExtensionError(f"Example docstring should have a header for the example title. "
325 f"Please check the example file:\n {script.script_file}\n")
326  
327 # Title is the first paragraph with any RST/Markdown title chars
328 # removed, i.e. lines that consist of (3 or more of the same) 7-bit
329 # non-ASCII chars.
330 # This conditional is not perfect but should hopefully be good enough.
331 title_paragraph = paragraphs[0]
332 match = FIRST_NON_MARKER_WITHOUT_HASH.search(title_paragraph)
333 if match is None:
334 raise ExtensionError(f"Could not find a title in first paragraph:\n{title_paragraph}")
335  
336 title = match.group(2).strip()
337  
338 # Use the title if no other paragraphs are provided
339 intro_paragraph = title if len(paragraphs) < 2 else paragraphs[1]
340  
341 # Concatenate all lines of the first paragraph
342 intro = re.sub('\n', ' ', intro_paragraph)
343 intro = _sanitize_md(intro)
344  
345 # Truncate at 95 chars
346 if len(intro) > 95:
347 intro = intro[:95] + '...'
348  
349 return title, intro

352 def create_thumb_from_image(script: GalleryScript, src_image_path: Path) -> Path:
353 """Create a thumbnail image from the `src_image_path`.
354  
355 Parameters
356 ----------
357 script : GalleryScript
358 The gallery script.
359  
360 src_image_path : Path
361 The source image path, with some flexibility about the extension.
362 TODO do we actually need this flexibility here ?
363  
364 Returns
365 -------
366 actual_thumb_file : Path
367 The actual thumbnail file generated.
368 """
369 try:
370 # Find the image, with flexibility about the actual extenstion ('png', 'svg', 'jpg', 'gif' are supported)
371 src_image_path, ext = _find_image_ext(src_image_path)
372 except ImageNotFoundError:
373 # The source image does not exist !
374 try:
375 # Does a thumbnail already exist ? with extenstion ('png', 'svg', 'jpg', 'gif')
376 thumb_file, ext = _find_image_ext(script.get_thumbnail_file(".png"))
377 # Yes - let's assume this one will suit the needs
378 return thumb_file
379 except ImageNotFoundError:
380 # Create something to replace the thumbnail
381 default_thumb_path = script.gallery_conf.get("default_thumb_file")
382 if default_thumb_path is None:
383 default_thumb_path = os.path.join(glr_path_static(), 'no_image.png')
384  
385 src_image_path, ext = _find_image_ext(Path(default_thumb_path))
386  
387 # Now let's create the thumbnail.
388 # - First Make sure the thumb dir exists
389 script.gallery.make_thumb_dir()
390  
391 # - Then create the thum file by copying the src image, possibly rescaling it.
392 thumb_file = script.get_thumbnail_file(ext)
393 if ext in ('.svg', '.gif'):
394 # No need to rescale image
395 copyfile(src_image_path, thumb_file)
396 else:
397 # Need to rescale image
398 max_width, max_hegiht = script.gallery_conf["thumbnail_size"]
399 rescale_image(in_file=src_image_path, out_file=thumb_file, max_width=max_width, max_height=max_hegiht)
400 if 'thumbnails' in script.gallery_conf['compress_images']:
401 optipng(thumb_file, script.gallery_conf['compress_images_args'])
402  
403 return thumb_file

406 def generate(gallery: GalleryBase, seen_backrefs: Set) -> Tuple[str, str, str, List[GalleryScriptResults]]:
407 """
408 Generate the gallery md for an example directory, including the index.
409  
410 Parameters
411 ----------
412 gallery : GalleryBase
413 The gallery or subgallery to process
414  
415 seen_backrefs : Set
416 Backrefs seen so far.
417  
418 Returns
419 -------
420 title : str
421 The gallery title, that is, the title of the readme file.
422  
423 root_subtitle : str
424 The gallery suptitle that will be used in case the gallery has subsections.
425  
426 index_md : str
427 The markdown to include in the global gallery readme.
428  
429 results : List[GalleryScriptResults]
430 A list of processing results for all scripts in this gallery.
431 """
432 # Read the gallery readme and add it to the index
433 readme_contents = gallery.readme_file.read_text(encoding="utf-8")
434 readme_title = extract_readme_title(gallery.readme_file, readme_contents)
435 if gallery.has_subsections():
436 # parse and try to also extract the last subtitle
437 last_readme_subtitle = extract_readme_last_subtitle(gallery.readme_file, readme_contents)
438 else:
439 # Dont look for the last subtitle
440 last_readme_subtitle = None
441  
442 # Create the destination dir if needed
443 gallery.make_generated_dir()
444  
445 all_thumbnail_entries = []
446 results = []
447  
448 for script in tqdm(gallery.scripts, desc=f"generating gallery for {gallery.generated_dir}... "):
449 # Generate all files related to this example: download file, jupyter notebook, pickle, markdown...
450 script_results = generate_file_md(script=script, seen_backrefs=seen_backrefs)
451 results.append(script_results)
452  
453 # Create the thumbnails-containing div <div class="mkd-glr-thumbcontainer" ...> to place in the readme
454 thumb_div = _thumbnail_div(script_results)
455 all_thumbnail_entries.append(thumb_div)
456  
457 # Write the gallery summary index.md
458 index_md = f"""<!-- {str(gallery.generated_dir_rel_project).replace(os.path.sep, '_')} -->
459  
460 {readme_contents}
461  
462 {"".join(all_thumbnail_entries)}
463 <div class="mkd-glr-clear"></div>
464  
465  
466 """
467 # Note: the "clear" is to disable floating elements again, now that the gallery section is over.
468  
469 return readme_title, last_readme_subtitle, index_md, results

476 def handle_exception(exc_info, script: GalleryScript):
477 """Trim and format exception, maybe raise error, etc."""
478 from .gen_gallery import _expected_failing_examples
479  
480 etype, exc, tb = exc_info
481 stack = traceback.extract_tb(tb)
482 # The full traceback will look something like:
483 #
484 # File "/home/larsoner/python/mkdocs-gallery/sphinx_gallery/gen_single.py...
485 # mem_max, _ = gallery_conf['call_memory'](
486 # File "/home/larsoner/python/mkdocs-gallery/sphinx_gallery/gen_galler...
487 # mem, out = memory_usage(func, max_usage=True, retval=True,
488 # File "/home/larsoner/.local/lib/python3.8/site-packages/memory_profi...
489 # returned = f(*args, **kw)
490 # File "/home/larsoner/python/mkdocs-gallery/sphinx_gallery/gen_single.py...
491 # exec(self.code, self.fake_main.__dict__)
492 # File "/home/larsoner/python/mkdocs-gallery/sphinx_gallery/tests/tiny...
493 # raise RuntimeError('some error')
494 # RuntimeError: some error
495 #
496 # But we should trim these to just the relevant trace at the user level,
497 # so we inspect the traceback to find the start and stop points.
498 start = 0
499 stop = len(stack)
500 root = os.path.dirname(__file__) + os.sep
501 for ii, s in enumerate(stack, 1):
502 # Trim our internal stack
503 if s.filename.startswith(root + 'gen_gallery.py') and s.name == 'call_memory':
504 start = max(ii, start)
505 elif s.filename.startswith(root + 'gen_single.py'):
506 # SyntaxError
507 if s.name == 'execute_code_block' and ('compile(' in s.line or 'save_figures' in s.line):
508 start = max(ii, start)
509 # Any other error
510 elif s.name == '__call__':
511 start = max(ii, start)
512 # Our internal input() check
513 elif s.name == '_check_input' and ii == len(stack):
514 stop = ii - 1
515 stack = stack[start:stop]
516  
517 formatted_exception = 'Traceback (most recent call last):\n' + ''.join(
518 traceback.format_list(stack) +
519 traceback.format_exception_only(etype, exc))
520  
521 src_file = script.src_py_file
522 expected = src_file in _expected_failing_examples(
523 gallery_conf=script.gallery_conf, mkdocs_conf=script.gallery.all_info.mkdocs_conf
524 )
525 if expected:
526 # func, color = logger.info, 'blue'
527 func = logger.info
528 else:
529 # func, color = logger.warning, 'red'
530 func = logger.warning
531 func(f"{src_file} failed to execute correctly: {formatted_exception}") # , color=color)
532  
533 except_md = codestr2md(formatted_exception, lang='pytb', is_exc=True)
534  
535 # Ensure it's marked as our style: this is now already done in codestr2md
536 # except_md = "{: .mkd-glr-script-out }\n\n" + except_md
537 return except_md, formatted_exception

558 """Patch warnings.showwarning to actually write out the warning."""

569 class _exec_once(object):
570 """Deal with memory_usage calling functions more than once (argh)."""
571  
572 def __init__(self, code, fake_main):
573 self.code = code
574 self.fake_main = fake_main
575 self.run = False
576  
577 def __call__(self):
578 if not self.run:
579 self.run = True
580 old_main = sys.modules.get('__main__', None)
581 with patch_warnings():
582 sys.modules['__main__'] = self.fake_main
583 try:
584 exec(self.code, self.fake_main.__dict__) # noqa # our purpose is to execute code :)
585 finally:
586 if old_main is not None:
587 sys.modules['__main__'] = old_main

590 def _get_memory_base(gallery_conf):
591 """Get the base amount of memory used by running a Python process."""
592 if not gallery_conf['plot_gallery']:
593 return 0.
594 # There might be a cleaner way to do this at some point
595 from memory_profiler import memory_usage
596 if sys.platform in ('win32', 'darwin'):
597 sleep, timeout = (1, 2)
598 else:
599 sleep, timeout = (0.5, 1)
  • S603 Subprocess call - check for execution of untrusted input.
600 proc = subprocess.Popen(
601 [sys.executable, '-c',

634 def _exec_and_get_memory(compiler, ast_Module, code_ast, script: GalleryScript):
635 """Execute ast, capturing output if last line is expression and get max memory usage."""
636  
637 src_file = script.src_py_file.as_posix()
638  
639 # capture output if last line is expression
640 is_last_expr = False
641  
642 if len(code_ast.body) and isinstance(code_ast.body[-1], ast.Expr):
643 is_last_expr = True
644 last_val = code_ast.body.pop().value
645 # exec body minus last expression
646 mem_body, _ = script.gallery_conf['call_memory'](
647 _exec_once(
648 compiler(code_ast, src_file, 'exec'),
649 script.run_vars.fake_main))
650 # exec last expression, made into assignment
651 body = [ast.Assign(targets=[ast.Name(id='___', ctx=ast.Store())], value=last_val)]
652 last_val_ast = ast_Module(body=body)
653 ast.fix_missing_locations(last_val_ast)
654 mem_last, _ = script.gallery_conf['call_memory'](
655 _exec_once(
656 compiler(last_val_ast, src_file, 'exec'),
657 script.run_vars.fake_main))
658 mem_max = max(mem_body, mem_last)
659 else:
660 mem_max, _ = script.gallery_conf['call_memory'](
661 _exec_once(
662 compiler(code_ast, src_file, 'exec'),
663 script.run_vars.fake_main))
664  
665 return is_last_expr, mem_max

668 def _get_last_repr(gallery_conf, ___):
669 """Get a repr of the last expression, using first method in 'capture_repr'
670 available for the last expression."""
671 for meth in gallery_conf['capture_repr']:
672 try:
673 last_repr = getattr(___, meth)()

739 def execute_code_block(compiler, block, script: GalleryScript):
740 """Execute the code block of the example file.
741  
742 Parameters
743 ----------
744 compiler : codeop.Compile
745 Compiler to compile AST of code block.
746  
747 block : List[Tuple[str, str, int]]
748 List of Tuples, each Tuple contains label ('text' or 'code'),
749 the corresponding content string of block and the leading line number.
750  
751 script: GalleryScript
752 The gallery script
753  
754 Returns
755 -------
756 code_output : str
757 Output of executing code in md.
758 """
759 # if script.run_vars.example_globals is None: # testing shortcut
760 # script.run_vars.example_globals = script.run_vars.fake_main.__dict__
761  
762 blabel, bcontent, lineno = block
763  
764 # If example is not suitable to run anymore, skip executing its blocks
765 if script.run_vars.stop_executing or blabel == 'text':
766 return ''
767  
768 cwd = os.getcwd()
769 # Redirect output to stdout
770 src_file = script.src_py_file
771 logging_tee = _check_reset_logging_tee(src_file)
772 assert isinstance(logging_tee, _LoggingTee) # noqa
773  
774 # First cd in the original example dir, so that any file
775 # created by the example get created in this directory
776 os.chdir(src_file.parent)
777  
778 sys_path = copy.deepcopy(sys.path)
779 sys.path.append(os.getcwd())
780  
781 # Save figures unless there is a `mkdocs_gallery_defer_figures` flag
782 match = re.search(r'^[\ \t]*#\s*mkdocs_gallery_defer_figures[\ \t]*\n?', bcontent, re.MULTILINE)
783 need_save_figures = match is None
784  
785 try:
786 ast_Module = _ast_module()
787 code_ast = ast_Module([bcontent])
788 flags = ast.PyCF_ONLY_AST | compiler.flags
789 code_ast = compile(bcontent, src_file, 'exec', flags, dont_inherit=1)
790 ast.increment_lineno(code_ast, lineno - 1)
791  
792 is_last_expr, mem_max = _exec_and_get_memory(compiler, ast_Module, code_ast, script=script)
793 script.run_vars.memory_used_in_blocks.append(mem_max)
794  
795 # This should be inside the try block, e.g., in case of a savefig error
796 logging_tee.restore_std()
797 if need_save_figures:
798 need_save_figures = False
799 images_md = save_figures(block, script)
800 else:
801 images_md = ""
802  
803 except Exception:
804 logging_tee.restore_std()
805 except_md, formatted_exception = handle_exception(sys.exc_info(), script)
806  
807 # Breaks build on first example error
808 if script.gallery_conf['abort_on_example_error']:
809 raise
810  
811 # Stores failing file
812 script.gallery_conf['failing_examples'][src_file] = formatted_exception
813  
814 # Stop further execution on that script
815 script.run_vars.stop_executing = True
816  
817 code_output = u"\n{0}\n\n\n\n".format(except_md)
818 # still call this even though we won't use the images so that
819 # figures are closed
820 if need_save_figures:
821 save_figures(block, script)
822 else:
823 _reset_cwd_syspath(cwd, sys_path)
824  
825 code_output = _get_code_output(is_last_expr, script, logging_tee, images_md)
826 finally:
827 _reset_cwd_syspath(cwd, sys_path)
828 logging_tee.restore_std()
829  
830 # Sanitize ANSI escape characters from MD output
831 ansi_escape = re.compile(r'\x1B(?:[@-Z\\-_]|\[[0-?]*[ -/]*[@-~])')
832 code_output = ansi_escape.sub('', code_output)
833  
834 return code_output

842 def parse_and_execute(script: GalleryScript, script_blocks):
843 """Execute and capture output from python script already in block structure
844  
845 Parameters
846 ----------
847 script : GalleryScript
848 The script
849  
850 script_blocks : list
851 (label, content, line_number)
852 List where each element is a tuple with the label ('text' or 'code'),
853 the corresponding content string of block and the leading line number
854  
855 Returns
856 -------
857 output_blocks : list
858 List of strings where each element is the restructured text
859 representation of the output of each block
860  
861 time_elapsed : float
862 Time elapsed during execution
863  
864 memory_used : float
865 Memory used during execution
866 """
867 # Examples may contain if __name__ == '__main__' guards for in example scikit-learn if the example uses
868 # multiprocessing. Here we create a new __main__ module, and temporarily change sys.modules when running our example
869 fake_main = importlib.util.module_from_spec(importlib.util.spec_from_loader('__main__', None))
870 script.run_vars.fake_main = fake_main
871  
872 example_globals = fake_main.__dict__
873 example_globals.update({
874 # A lot of examples contains 'print(__doc__)' for example in
875 # scikit-learn so that running the example prints some useful
876 # information. Because the docstring has been separated from
877 # the code blocks in mkdocs-gallery, __doc__ is actually
878 # __builtin__.__doc__ in the execution context and we do not
879 # want to print it
880 '__doc__': '',
881 # Don't ever support __file__: Issues #166 #212
882 # Don't let them use input()
883 'input': _check_input,
884 })
885 script.run_vars.example_globals = example_globals
886  
887 # Manipulate the sys.argv before running the example
888 # See https://github.com/sphinx-gallery/sphinx-gallery/pull/252
889  
890 # Remember the original argv so that we can put them back after run
891 argv_orig = sys.argv[:]
892  
893 # Python file is the original one (not the copy for download)
894 sys.argv[0] = script.src_py_file.as_posix()
895  
896 # Allow users to provide additional args through the 'reset_argv' option
897 sys.argv[1:] = script.gallery_conf['reset_argv'](script)
898  
899 # Perform a garbage collection before starting so that perf kpis are accurate (memory and time)
900 gc.collect()
901  
902 # Initial memory used
903 memory_start, _ = script.gallery_conf['call_memory'](lambda: None)
904 script.run_vars.memory_used_in_blocks = [memory_start] # include at least one entry to avoid max() ever failing
905  
906 t_start = time()
907 compiler = codeop.Compile()
908  
909 # Execute block by block
910 output_blocks = list()
911 with _LoggingTee(script.src_py_file) as logging_tee:
912 for block in script_blocks:
913 logging_tee.set_std_and_reset_position()
914 output_blocks.append(execute_code_block(compiler, block, script))
915  
916 # Compute the elapsed time
917 time_elapsed = time() - t_start
918  
919 # Set back the sys argv
920 sys.argv = argv_orig
921  
922 # Write md5 checksum if the example was meant to run (no-plot shall not cache md5sum) and has built correctly
923 script.write_final_md5_file()
924  
925 # Declare the example as "passing"
926 script.gallery_conf['passing_examples'].append(script)
927  
928 script.run_vars.memory_delta = max(script.run_vars.memory_used_in_blocks) - memory_start
929 memory_used = script.gallery_conf['memory_base'] + script.run_vars.memory_delta
930  
931 return output_blocks, time_elapsed, memory_used

1102 def generate_md_from_blocks(script_blocks, output_blocks, file_conf, gallery_conf) -> str:
