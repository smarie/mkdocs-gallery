69 def rst2md(text, gallery_conf, target_dir, heading_levels):
70 """Converts the RST text from the examples docstrings and comments
71 into markdown text for the Jupyter notebooks
72  
73 Parameters
74 ----------
75 text: str
76 RST input to be converted to MD
77 gallery_conf : dict
78 The mkdocs-gallery configuration dictionary.
79 target_dir : str
80 Path that notebook is intended for. Used where relative paths
81 may be required.
82 heading_levels: dict
83 Mapping of heading style ``(over_char, under_char)`` to heading level.
84 Note that ``over_char`` is `None` when only underline is present.
85 """
86  
87 # Characters recommended for use with headings
88 # https://docutils.readthedocs.io/en/sphinx-docs/user/rst/quickstart.html#sections
89 adornment_characters = "=`:.'\"~^_*+#<>-"
90 headings = re.compile(
91 # Start of string or blank line
92 r'(?P<pre>\A|^[ \t]*\n)'
93 # Optional over characters, allowing leading space on heading text
94 r'(?:(?P<over>[{0}])(?P=over)*\n[ \t]*)?'
95 # The heading itself, with at least one non-white space character
96 r'(?P<heading>\S[^\n]*)\n'
97 # Under character, setting to same character if over present.
98 r'(?P<under>(?(over)(?P=over)|[{0}]))(?P=under)*$'
99 r''.format(adornment_characters),
100 flags=re.M)
101  
102 text = re.sub(
103 headings,
104 lambda match: '{1}{0} {2}'.format(
105 '#'*heading_levels[match.group('over', 'under')],
106 *match.group('pre', 'heading')),
107 text)
108  
109 math_eq = re.compile(r'^\.\. math::((?:.+)?(?:\n+^ .+)*)', flags=re.M)
110 text = re.sub(math_eq,
111 lambda match: r'\begin{{align}}{0}\end{{align}}'.format(
112 match.group(1).strip()),
113 text)
114 inline_math = re.compile(r':math:`(.+?)`', re.DOTALL)
115 text = re.sub(inline_math, r'$\1$', text)
116  
117 directives = ('warning', 'note')
118 for directive in directives:
119 directive_re = re.compile(r'^\.\. %s::((?:.+)?(?:\n+^ .+)*)'
120 % directive, flags=re.M)
121 text = re.sub(directive_re,
122 partial(directive_fun, directive=directive), text)
123  
124 links = re.compile(r'^ *\.\. _.*:.*$\n', flags=re.M)
125 text = re.sub(links, '', text)
126  
127 refs = re.compile(r':ref:`')
128 text = re.sub(refs, '`', text)
129  
130 contents = re.compile(r'^\s*\.\. contents::.*$(\n +:\S+: *$)*\n',
131 flags=re.M)
132 text = re.sub(contents, '', text)
133  
134 images = re.compile(
135 r'^\.\. image::(.*$)((?:\n +:\S+:.*$)*)\n',
136 flags=re.M)
137 image_opts = re.compile(r'\n +:(\S+): +(.*)$', flags=re.M)
138 text = re.sub(
139 images,
140 lambda match: '<img src="{}"{}>\n'.format(
141 generate_image_src(
142 match.group(1).strip(), gallery_conf, target_dir),
143 re.sub(image_opts, r' \1="\2"', match.group(2) or '')),
144 text)
145  
146 return text

181 def jupyter_notebook(script: GalleryScript, script_blocks: List):
182 """Generate a Jupyter notebook file cell-by-cell
183  
184 Parameters
185 ----------
186 script : GalleryScript
187 Script
188  
189 script_blocks : list
190 Script execution cells.
191 """
192 # Grab the possibly custom first and last cells
193 first_cell = script.gallery_conf["first_notebook_cell"]
194 last_cell = script.gallery_conf["last_notebook_cell"]
195  
196 # Initialize with a notebook skeleton
197 work_notebook = jupyter_notebook_skeleton()
198  
199 # Custom first cell
200 if first_cell is not None:
201 add_code_cell(work_notebook, first_cell)
202  
203 # Fill the notebook per se
204 fill_notebook(work_notebook, script_blocks)
205  
206 # Custom last cell
207 if last_cell is not None:
208 add_code_cell(work_notebook, last_cell)
209  
210 return work_notebook

221

224 "execution_count": None,
225 "metadata": {"collapsed": False},
226 "outputs": [],
227 "source": [code.strip()]
228 }
229 work_notebook["cells"].append(code_cell)
