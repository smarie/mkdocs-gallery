26 def gen_binder_url(script: GalleryScript, binder_conf):
27 """Generate a Binder URL according to the configuration in conf.py.
28  
29 Parameters
30 ----------
31 script: GalleryScript
32 The script for which a Binder badge will be generated.
33 binder_conf: dict or None
34 The Binder configuration dictionary. See `gen_binder_md` for details.
35  
36 Returns
37 -------
38 binder_url : str
39 A URL that can be used to direct the user to the live Binder environment.
40 """
41 # Build the URL
42 fpath_prefix = binder_conf.get('filepath_prefix')
43 link_base = binder_conf.get('notebooks_dir')
44  
45 # We want to keep the relative path to sub-folders
46 path_link = os.path.join(link_base, script.ipynb_file_rel_site_root.as_posix())
47  
48 # In case our website is hosted in a sub-folder
49 if fpath_prefix is not None:
50 path_link = '/'.join([fpath_prefix.strip('/'), path_link])
51  
52 # Make sure we have the right slashes (in case we're on Windows)
53 path_link = path_link.replace(os.path.sep, '/')
54  
55 # Create the URL
56 # See https://mybinder.org/ to check that it is still the right one
57 # Note: the branch will typically be gh-pages
58 binder_url = '/'.join([binder_conf['binderhub_url'],
59 'v2', 'gh',
60 binder_conf['org'],
61 binder_conf['repo'],
62 quote(binder_conf['branch'])])
63  
64 if binder_conf.get('use_jupyter_lab', False) is True:
65 binder_url += '?urlpath=lab/tree/{}'.format(quote(path_link))
66 else:
67 binder_url += '?filepath={}'.format(quote(path_link))
68 return binder_url

71 def gen_binder_md(script: GalleryScript, binder_conf: Dict):
72 """Generate the MD + link for the Binder badge.
73  
74 Parameters
75 ----------
76 script: GalleryScript
77 The script for which a Binder badge will be generated.
78  
79 binder_conf: dict or None
80 If a dictionary it must have the following keys:
81  
82 'binderhub_url'
83 The URL of the BinderHub instance that's running a Binder service.
84 'org'
85 The GitHub organization to which the documentation will be pushed.
86 'repo'
87 The GitHub repository to which the documentation will be pushed.
88 'branch'
89 The Git branch on which the documentation exists (e.g., gh-pages).
90 'dependencies'
91 A list of paths to dependency files that match the Binderspec.
92  
93 Returns
94 -------
95 md : str
96 The Markdown for the Binder badge that links to this file.
97 """
98 binder_url = gen_binder_url(script, binder_conf)
99  
100 # TODO revisit this comment for mkdocs
101 # In theory we should be able to use glr_path_static for this, but Sphinx only allows paths to be relative to the
102 # build root. On Linux, absolute paths can be used and they work, but this does not seem to be
103 # documented behavior: https://github.com/sphinx-doc/sphinx/issues/7772
104 # And in any case, it does not work on Windows, so here we copy the SVG to `images` for each gallery and link to it
105 # there. This will make a few copies, and there will be an extra in `_static` at the end of the build, but it at
106 # least works...
107 physical_path = script.gallery.images_dir / "binder_badge_logo.svg"
108 if not physical_path.exists():
109 # Make sure parent dirs exists (this should not be necessary actually)
110 physical_path.parent.mkdir(parents=True, exist_ok=True)
111 shutil.copyfile(os.path.join(glr_path_static(), 'binder_badge_logo.svg'), str(physical_path))
112 else:
113 assert physical_path.is_file() # noqa
114  
115 # Create the markdown image with a link
116 return f"[![Launch binder](./images/binder_badge_logo.svg)]({binder_url}){{ .center}}"

138 def _copy_binder_reqs(binder_conf, mkdocs_conf):
139 """Copy Binder requirements files to a ".binder" folder in the docs.
140  
141 See https://mybinder.readthedocs.io/en/latest/using/config_files.html#config-files
142 """
143 path_reqs = binder_conf.get('dependencies')
144  
145 # Check that they exist (redundant since the check is already done by mkdocs.)
146 for path in path_reqs:
147 if not os.path.exists(path):
148 raise ConfigError(f"Couldn't find the Binder requirements file: {path}, did you specify it correctly?")
149  
150 # Destination folder: a ".binder" folder
151 binder_folder = os.path.join(mkdocs_conf['site_dir'], '.binder')
152 if not os.path.isdir(binder_folder):
153 os.makedirs(binder_folder)
154  
155 # Copy over the requirement files to the output directory
156 for path in path_reqs:
157 shutil.copy(path, binder_folder)

160 def _remove_ipynb_files(path, contents):
161 """Given a list of files in `contents`, remove all files named `ipynb` or
162 directories named `images` and return the result.
163  
164 Used with the `shutil` "ignore" keyword to filter out non-ipynb files."""
165 contents_return = []
166 for entry in contents:
167 if entry.endswith('.ipynb'):
168 # Don't include ipynb files
169 pass
170 elif (entry != "images") and os.path.isdir(os.path.join(path, entry)):
171 # Don't include folders not called "images"
172 pass
173 else:
174 # Keep everything else
175 contents_return.append(entry)
176 return contents_return

199 def check_binder_conf(binder_conf):
200 """Check to make sure that the Binder configuration is correct."""
201  
202 # Grab the configuration and return None if it's not configured
203 binder_conf = {} if binder_conf is None else binder_conf
204 if not isinstance(binder_conf, dict):
205 raise ConfigError('`binder_conf` must be a dictionary or None.')
206 if len(binder_conf) == 0:
207 return binder_conf
208  
209 # Ensure all fields are populated
210 req_values = ['binderhub_url', 'org', 'repo', 'branch', 'dependencies']
211 optional_values = ['filepath_prefix', 'notebooks_dir', 'use_jupyter_lab']
212 missing_values = []
213 for val in req_values:
214 if binder_conf.get(val) is None:
215 missing_values.append(val)
216  
217 if len(missing_values) > 0:
218 raise ConfigError(f"binder_conf is missing values for: {missing_values}")
219  
220 for key in binder_conf.keys():
221 if key not in (req_values + optional_values):
222 raise ConfigError(f"Unknown Binder config key: {key}")
223  
224 # Ensure we have http in the URL
225 if not any(binder_conf['binderhub_url'].startswith(ii) for ii in ['http://', 'https://']):
226 raise ConfigError(f"did not supply a valid url, gave binderhub_url: {binder_conf['binderhub_url']}")
227  
228 # Ensure we have at least one dependency file
229 # Need at least one of these three files
230 required_reqs_files = ['requirements.txt', 'environment.yml', 'Dockerfile']
231  
232 path_reqs = binder_conf['dependencies']
233 if isinstance(path_reqs, str):
234 path_reqs = [path_reqs]
235 binder_conf['dependencies'] = path_reqs
236 elif not isinstance(path_reqs, (list, tuple)):
237 raise ConfigError(f"`dependencies` value should be a list of strings. Got type {type(path_reqs)}.")
238  
239 binder_conf['notebooks_dir'] = binder_conf.get('notebooks_dir', 'notebooks')
240  
241 path_reqs_filenames = [os.path.basename(ii) for ii in path_reqs]
242 if not any(ii in path_reqs_filenames for ii in required_reqs_files):
243 raise ConfigError("Did not find one of `requirements.txt` or `environment.yml` in the \"dependencies\" section"
244 " of the binder configuration for mkdocs-gallery. A path to at least one of these files must"
245 " exist in your Binder dependencies.")
246  
247 return binder_conf
