47 class NameFinder(ast.NodeVisitor):
48 """Finds the longest form of variable names and their imports in code.
49  
50 Only retains names from imported modules.
51 """
52  
53 def __init__(self, global_variables=None):
54 super(NameFinder, self).__init__()
55 self.imported_names = {}
56 self.global_variables = global_variables or {}
57 self.accessed_names = set()
58  
59 def visit_Import(self, node, prefix=''):
60 for alias in node.names:
61 local_name = alias.asname or alias.name
62 self.imported_names[local_name] = prefix + alias.name
63  
64 def visit_ImportFrom(self, node):
65 self.visit_Import(node, node.module + '.')
66  
67 def visit_Name(self, node):
68 self.accessed_names.add(node.id)
69  
70 def visit_Attribute(self, node):
71 attrs = []
72 while isinstance(node, ast.Attribute):
73 attrs.append(node.attr)
74 node = node.value
75  
76 if isinstance(node, ast.Name):
77 # This is a.b, not e.g. a().b
78 attrs.append(node.id)
79 self.accessed_names.add('.'.join(reversed(attrs)))
80 else:
81 # need to get a in a().b
82 self.visit(node)

84 def get_mapping(self):
85 options = list()
86 for name in self.accessed_names:
87 local_name_split = name.split('.')
88 # first pass: by global variables and object inspection (preferred)
89 for split_level in range(len(local_name_split)):
90 local_name = '.'.join(local_name_split[:split_level + 1])
91 remainder = name[len(local_name):]
92 if local_name in self.global_variables:
93 obj = self.global_variables[local_name]
94 class_attr, method = False, []
95 if remainder:
96 for level in remainder[1:].split('.'):
97 last_obj = obj
98 # determine if it's a property
99 prop = getattr(last_obj.__class__, level, None)
100 if isinstance(prop, property):
101 obj = last_obj
102 class_attr, method = True, [level]
103 break
104 try:
105 obj = getattr(obj, level)
106 except AttributeError:
107 break
108 if inspect.ismethod(obj):
109 obj = last_obj
110 class_attr, method = True, [level]
111 break
112 del remainder
113 is_class = inspect.isclass(obj)
114 if is_class or class_attr:
115 # Traverse all bases
116 classes = [obj if is_class else obj.__class__]
117 offset = 0
118 while offset < len(classes):
119 for base in classes[offset].__bases__:
120 # "object" as a base class is not very useful
121 if base not in classes and base is not object:
122 classes.append(base)
123 offset += 1
124 else:
125 classes = [obj.__class__]
126 for cc in classes:
127 module = inspect.getmodule(cc)
128 if module is not None:
129 module = module.__name__.split('.')
130 class_name = cc.__qualname__
131 # a.b.C.meth could be documented as a.C.meth,
132 # so go down the list
133 for depth in range(len(module), 0, -1):
134 full_name = '.'.join(
135 module[:depth] + [class_name] + method)
136 options.append(
137 (name, full_name, class_attr, is_class))
138 # second pass: by import (can't resolve as well without doing
139 # some actions like actually importing the modules, so use it
140 # as a last resort)
141 for split_level in range(len(local_name_split)):
142 local_name = '.'.join(local_name_split[:split_level + 1])
143 remainder = name[len(local_name):]
144 if local_name in self.imported_names:
145 full_name = self.imported_names[local_name] + remainder
146 is_class = class_attr = False # can't tell without import
147 options.append(
148 (name, full_name, class_attr, is_class))
149 return options

152 def _from_import(a, b):
153 # imp_line = 'from %s import %s' % (a, b)
154 # scope = dict()
155 # with warnings.catch_warnings(record=True): # swallow warnings
156 # warnings.simplefilter('ignore')
157 # exec(imp_line, scope, scope)
158 # return scope
159 with warnings.catch_warnings(record=True): # swallow warnings
160 warnings.simplefilter('ignore')
161 m = import_module(a)
162 obj = getattr(m, b)

164 return obj

167 def _get_short_module_name(module_name, obj_name):
168 """Get the shortest possible module name."""
169 if '.' in obj_name:
170 obj_name, attr = obj_name.split('.')
171 else:
172 attr = None
173 # scope = {}
174 try:
175 # Find out what the real object is supposed to be.
176 imported_obj = _from_import(module_name, obj_name)
177 except Exception: # wrong object
178 return None
179 else:
180 real_obj = imported_obj
181 if attr is not None and not hasattr(real_obj, attr): # wrong class
182 return None # wrong object
183  
184 parts = module_name.split('.')
185 short_name = module_name
186 for i in range(len(parts) - 1, 0, -1):
187 short_name = '.'.join(parts[:i])
188 # scope = {}
189 try:
190 imported_obj = _from_import(short_name, obj_name)
191 # Ensure shortened object is the same as what we expect.
192 assert real_obj is imported_obj # noqa
193 except Exception: # libraries can throw all sorts of exceptions...
194 # get the last working module name
195 short_name = '.'.join(parts[:(i + 1)])
196 break
197 return short_name

209 def identify_names(script_blocks, global_variables=None, node=''):
210 """Build a codeobj summary by identifying and resolving used names."""
211  
212 if node == '': # mostly convenience for testing functions
213 c = '\n'.join(txt for kind, txt, _ in script_blocks if kind == 'code')
214 node = ast.parse(c)
215  
216 # Get matches from the code (AST)
217 finder = NameFinder(global_variables)
218 if node is not None:
219 finder.visit(node)
220 names = list(finder.get_mapping())
221  
222 # Get matches from docstring inspection
223 text = '\n'.join(txt for kind, txt, _ in script_blocks if kind == 'text')
224 names.extend((x, x, False, False) for x in re.findall(_regex, text))
225 example_code_obj = collections.OrderedDict() # order is important
226  
227 # Make a list of all guesses, in `_embed_code_links` we will break when we find a match
228 for name, full_name, class_like, is_class in names:
229 if name not in example_code_obj:
230 example_code_obj[name] = list()
231  
232 # name is as written in file (e.g. np.asarray)
233 # full_name includes resolved import path (e.g. numpy.asarray)
234 splitted = full_name.rsplit('.', 1 + class_like)
235 if len(splitted) == 1:
236 splitted = ('builtins', splitted[0])
237 elif len(splitted) == 3: # class-like
238 assert class_like # noqa
239 splitted = (splitted[0], '.'.join(splitted[1:]))
240 else:
241 assert not class_like # noqa
242  
243 module, attribute = splitted
244  
245 # get shortened module name
246 module_short = _get_short_module_name(module, attribute)
247 cobj = {'name': attribute, 'module': module,
248 'module_short': module_short or module, 'is_class': is_class}
249  
250 example_code_obj[name].append(cobj)
251  
252 return example_code_obj

318 """
319 Write backreference file including a thumbnail list of examples.
320  
321 Parameters
322 ----------
323 backrefs : set
324  
325 seen_backrefs : set
326  
327 script_results : GalleryScriptResults
328  
329 Returns
330 -------
331  
332 """
333 all_info = script_results.script.gallery.all_info
334  
335 for backref in backrefs:
336 # Get the backref file to use for this module, according to config
337 include_path = _new_file(all_info.get_backreferences_file(backref))
338  
339 # Create new or append to existing file
340 seen = backref in seen_backrefs
341 with codecs.open(str(include_path), 'a' if seen else 'w', encoding='utf-8') as ex_file:
342 # If first ref: write header
343 if not seen:
344 # Be aware that if the number of lines of this heading changes,
345 # the minigallery directive should be modified accordingly
346 heading = 'Examples using ``%s``' % backref
347 ex_file.write('\n\n' + heading + '\n')
348 ex_file.write('^' * len(heading) + '\n')
349  
350 # Write the thumbnail
351 ex_file.write(_thumbnail_div(script_results, is_backref=True))
352 seen_backrefs.add(backref)
